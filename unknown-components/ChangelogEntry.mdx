export const ChangelogEntry = ({ version, date, type, changes, breaking }) => {
  const typeColors = {
    major: 'bg-red-100 text-red-800 border-red-200',
    minor: 'bg-blue-100 text-blue-800 border-blue-200',
    patch: 'bg-green-100 text-green-800 border-green-200',
  };
  return (
    <div className="border-l-4 border-blue-400 pl-6 pb-8 relative">
      <div className="absolute -left-2 top-0 w-4 h-4 bg-blue-400 rounded-full" />

        <span className="font-bold text-xl">{version}</span>
        <span className={`px-2 py-0.5 rounded-full text-xs font-medium border ${typeColors[type] || typeColors.patch}`}>
          {type}
        </span>l
        <span className="text-gray-400 text-sm">{date}</span>
      </div>
      {breaking && breaking.length > 0 && (
        <div className="bg-red-50 border border-red-200 rounded p-3 mb-3">
          <span className="font-bold text-red-700">⚠️ Breaking Changes:</span>
          <ul className="mt-1 mb-0 text-sm">
            {breaking.map((b, i) => <li key={i} className="text-red-600">{b}</li>)}
          </ul>
        </div>
      )}
      <ul className="space-y-1">
        {changes.map((c, i) => <li key={i}>{c}</li>)}
      </ul>
    </div>
  );
};
