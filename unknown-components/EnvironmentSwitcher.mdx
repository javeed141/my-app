export const EnvironmentSwitcher = ({ environments }) => {
  const [active, setActive] = useState(0);
  const env = environments[active];
  return (
    <div className="border rounded-lg overflow-hidden my-6">
      <div className="flex border-b">
        {environments.map((e, i) => (
          <button
            key={i}
            onClick={() => setActive(i)}
            className={`px-6 py-3 font-medium text-sm ${
              i === active ? 'bg-blue-500 text-white' : 'bg-gray-100 text-gray-600 hover:bg-gray-200'
            }`}
          >
            {e.name}
          </button>
        ))}
      </div>
      <div className="p-6">
        <div className="font-mono text-sm bg-gray-100 p-3 rounded mb-3">{env.baseUrl}</div>

        <div className="mt-3 text-xs text-gray-400">API Key prefix: <code>{env.keyPrefix}</code></div>
      </div>
    </div>
  );
};
